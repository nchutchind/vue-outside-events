v-mouseover-outside example

		

		<script>
			export default {
				methods: {
					onMouseOverOutside(e, el) {
						console.log("onMouseOverOutside");
						console.log("Event:", e);
						console.log("Element mouse moved over:", e.target);
						console.log("Element that leered from behind other elements:", el);
					}
				}
			}
		</script>
